        document.getElementById('timer').innerText = `${m<10?'0':''}${m}:${s<10?'0':''}${s}`;
    }, 1000);
}

function loadQuestion() {
    const q = questions[currentIdx];
    document.getElementById('qText').innerText = `${currentIdx + 1}. ${q.q}`;
    const box = document.getElementById('optionsBox');
    box.innerHTML = '';
    
    // Жауаптарды көрсету (таңдалған болса белгілеу)
    const pfx = ['A', 'B', 'C', 'D'];
    q.a.forEach((opt, i) => {
        const div = document.createElement('div');
        div.className = 'option' + (userAnswers[currentIdx] === opt ? ' selected' : '');
        div.innerHTML = `<span class="prefix">${pfx[i]})</span> ${opt}`;
        div.onclick = () => selectOption(opt);
        box.appendChild(div);
    });

    document.getElementById('progress').innerText = `Сұрақ: ${currentIdx + 1} / ${questions.length}`;
    document.getElementById('prevBtn').disabled = currentIdx === 0;
}

function selectOption(opt) {
    userAnswers[currentIdx] = opt;
    loadQuestion();
}

function nextQuestion() {
    if(currentIdx < questions.length - 1) {
        currentIdx++;
        loadQuestion();
    }
}

function prevQuestion() {
    if(currentIdx > 0) {
        currentIdx--;
        loadQuestion();
    }
}

function finishQuiz() {
    if(!confirm("Тестті аяқтауға сенімдісіз бе?")) return;
    
    document.getElementById('quiz-ui').style.display = 'none';
    document.getElementById('result-ui').style.display = 'block';
    
    let score = 0;
    const tableBody = document.getElementById('reviewBody');
    
    questions.forEach((q, i) => {
        const uAns = userAnswers[i] || "Жауап жоқ";
        const isCorrect = uAns === q.c;
        if(isCorrect) score++;

        const row = document.createElement('tr');
        if(!isCorrect) row.className = 'wrong-row';
        row.innerHTML = `
            <td>${q.q}</td>
            <td style="color:${isCorrect?'#27ae60':'#e74c3c'}">${uAns}</td>
            <td style="color:#27ae60; font-weight:bold;">${q.c}</td>
        `;
        tableBody.appendChild(row);
    });

    document.getElementById('finalScore').innerText = `Нәтиже: ${score} / ${questions.length}`;
}
